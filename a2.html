# load worldbank data and return two dataframes: one with years as columns and one with countries as columns.
# for specific indicators
def load_format_data(filename):
    df = pd.read_csv(filename, skiprows=4)

    df = df.drop(columns=['Country Code',
                          'Indicator Code',
                          'Unnamed: 67'])
    df = df.rename(columns={'Country Name': 'Country',
                            'Indicator Name': 'Indicator'})

    # select specific countries
    countries = ['Canada', 'China', 'France',
                 'Germany', 'Japan', 'India',
                 'Italy', 'United States', 'United Kingdom']
    df = df.loc[df['Country'].isin(countries)]

    # select specific indicators
    indicators = ['CO2 emissions (kt)',
                  'Agricultural land (sq. km)',
                  'Forest area (sq. km)',
                  'Population growth (annual %)']
    df_selected = df.loc[df['Indicator'].isin(indicators)]

    # Identify columns with numeric data
    numeric_columns = df.select_dtypes(include=['number']).columns
    # Fill missing values in each row with the mean of that row for numeric columns only
    df_selected[numeric_columns] = df_selected[numeric_columns].apply(lambda row: row.fillna(row.mean()), axis=1)

    # reset index
    df_selected.reset_index(drop=True, inplace=True)

    # Melt DataFrame to convert years into rows
    df_melted = df_selected.melt(id_vars=['Country', 'Indicator'],
                                 var_name='Year', value_name='Value')

    # Pivot the melted DataFrame
    df_pivoted = df_melted.pivot_table(index=['Year', 'Indicator'],
                                       columns='Country', values='Value').reset_index()

    # Remove the name of the index
    indicator_by_years = df_pivoted.rename_axis(None, axis=1)

    # return two dataframes
    return indicator_by_years, df_selected
